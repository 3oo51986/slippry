jQuery

jQuery('#news-demo').slippry({
  // general elements & wrapper
  slippryWrapper: '<div class="slippry_box news-slider" />',
  elements: 'article',

  // options
  adaptHeight: false,

  // pager
  pagerClass: 'news-pager',

  // transitions
  transition: 'horizontal',
  useCSS: true,

  // slideshow
  autoDirection: 'prev'
});
