jQuery

jQuery('#shop-demo').slippry({
  // general elements & wrapper
  slippryWrapper: '<div class="slippry_box shop-slider" />',
  elements: 'article',

  // options
  adaptiveHeight: false,
  start: 2,
  loop: false,
  captions: 'custom',
  captionsEl: '.product-name',

  // pager
  pager: false,

  // transitions
  slideMargin: 20,

  //slideshow
  auto: false,
});
